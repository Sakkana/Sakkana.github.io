如何使用 go mod 管理包

之前使用 go get 下载源码总是出问题，后来学习了一下 go 管理工具的知识才明白一点。
总的来说 go 经历了三代包管理工具的优化，gopath，vendor 和现在的 go mod。
这里使用 go mod 管理，提前要把 go env 中的 go1111nmoudle 设置为 on。
进入要使用某个框架的目录，执行 go mod init，此时会创建一个名为 go.mod 的文件。
现在使用 go get 拉去框架后会把依赖的信息自动添加到这个文件中。

下面的笔记部分来自于 go-tour 和开源文档《go语言圣经》（ “go pragramming language” 的民间翻译版本，质量颇高）。

变量

命名

var 声明变量列表，类型在最后
经常用于包和函数级别的变量

package main

import "fmt"

var a, b, c int
var d, e float64
var x string

func main() {
	a, b, c = 1, 2, 3
	d, e, x = 1.1, 2.2, "happy"
	fmt.Println(a, b, c, d, e, x)
}

如果变量的声明没有初始值，必须要显式写出类型

package main

import "fmt"

func main() {
	var a int
	fmt.Println(a)
}

如果变量有了初始值，可以省略类型，进行自动推导

package main

import "fmt"

func main() {
	var a, b, c = 1, true, "happy"
	fmt.Println(a, b, c)
}

海象符 := 可以代替 var，但是只能用于函数体内部

package main

import "fmt"

func main() {
	x := 1
	a, b, c := "123", false, 12.0
	fmt.Println(x, a, b, c)
}

结构体
package main

	X int
	Y int

Go 的结构体指针和 C++ 略有不同
C++ 需要用 (*p).X 或者 p->X 来访问，
但是 Go 简化了，只用 p.X 就可以访问了。

package main

import "fmt"

type Vertex struct {
	X int
	Y int
}

func main() {
	a := Vertex{1, 2}
	p := &a
	fmt.Println(p.X, p.Y)
}

结构体可以被显式指定哪个字段被赋值

package main

切片的长度与容量

package main

import "fmt"

func main() {
	s := []int{2, 3, 5, 7, 11, 13}
	printSlice(s)

	// 截取切片使其长度为 0
	s = s[:0]
	printSlice(s)

	// 拓展其长度
	s = s[:4]
	printSlice(s)

	// 舍弃前一个值
	s = s[1:4]
	printSlice(s)
}

func printSlice(s []int) {
	fmt.Printf("len=%d cap=%d %v\n", len(s), cap(s), s)
}

len=0 cap=6 []
len=4 cap=6 [2 3 5 7]
len=3 cap=5 [3 5 7]

映射

一个 map 被表示成 map[K]V，这个概念 C++、Java、Python 啥的也都有。

package main

import "fmt"

func main() {
	// 定义并初始化
	var m1 map[int]string = map[int]string{12: "Bob", 1: "Alice", 45: "Andy"}
	fmt.Println(m1)

	// 定义但不初始化
	var m2 map[int]string
	m2 = make(map[int]string) // 必须要这个，不然会报错，因为给一个 nil map 赋值了
	m2[1] = "Bob"
	m2[2] = "Alice"
	fmt.Println(m2)

	// 用海象符定义并创建
	m3 := make(map[int]string)
	m3[0] = "fuck"
	fmt.Println(m3)
}

增删查改

package main

import "fmt"

func main() {
    var m map[int]string = make(map[int]string)

    // 插入元素
    m[0] = "Alice"
    m[1] = "Bob"
    fmt.Println(m)

    // 删除元素
    delete(m, 0)
    fmt.Println(m)

    // 获取
    elem := m[1]
    fmt.Println(elem)

    // 检测是否存在
    elem, ok := m[1]
    fmt.Printf("elem = %v, ok = %v", elem, ok)
}

练习：映射
实现 WordCount。它应当返回一个映射，其中包含字符串 s 中每个“单词”的个数。函数 wc.Test 会对此函数执行一系列测试用例，并输出成功还是失败。
strings.Fields 会把字符串以空格分割，存在一个 string 数组里面。

package main

import (
	"fmt"
	"strings"
)

func main() {
	s := "The quick brown fox jumped over the lazy lazy dog."
	stringArr := strings.Fields(s)
	fmt.Printf("type = %T\n", stringArr)
}

答案

package main

import (
	"golang.org/x/tour/wc"
	"strings"
)

func WordCount(s string) map[string]int {
	m := make(map[string]int)
	stringArr := strings.Fields(s)
	for _, v := range stringArr {
		m[v]++
	}
	return m
}

func main() {
	wc.Test(WordCount)
}

函数的闭包
Go 函数可以是一个闭包。闭包是一个函数值，它引用了其函数体之外的变量。该函数可以访问并赋予其引用的变量的值，换句话说，该函数被这些变量“绑定”在一起。

例如，函数 adder 返回一个闭包。每个闭包都被绑定在其各自的 sum 变量上。

package main

import "fmt"

func adder() func(int) int {
	sum := 0
	return func(x int) int {
		sum += x
		return sum
	}
}

func main() {
	pos, neg := adder(), adder()
	for i := 0; i < 10; i++ {
		fmt.Println(
			pos(i),
			neg(-2*i),
		)
	}
}

函数值不仅仅是一串代码，还记录了状态。

package main

import "fmt"

// 返回一个“返回int的函数”
func fibonacci() func() int {
	x, y := 0, 1
	return func() int {
		t := x 
		x, y = y, x + y
		return t
	}
}

func main() {
	f := fibonacci()
	for i := 0; i < 10; i++ {
		fmt.Println(f())
	}
}

流程控制

for 循环

for 循环是 Go 唯一一种循环结构
和 C++，Java 非常不同，for 后面没有小括号，但是必须要有大括号

package main

import "fmt"

func main() {
	fmt.Println("prinmes:")
	for i := 1; i <= 10; i++ {
		fmt.Println(i)
	}
}

初始值和步长都是可选的

package main

import "fmt"

func main() {
	i := 1
	for ;i <= 10;{

Go 允许在 if 的条件之前执行一句简短的语句。

		return x

牛顿法计算平方根

package main

import (
    "fmt"
    "math"
)

func Sqrt(x float64) float64 {
    z := 1.0
    for i := 0; i < 10; i++ {
        z -= (z*z - x) / (2 * z)
        fmt.Println(z)
    }
    return z
}

func main() {
    stdres := math.Sqrt(2)
    myres := Sqrt(2)
    fmt.Printf("标准库答案：%v\n", stdres)
    fmt.Printf("我的答案：%v\n", myres)
    fmt.Printf("误差：%v\n", stdres-myres)
}
switch

Go 的 switch 自动在每个 case 后面添加了 break，只要走完一个 case 马上退出。
这是和 C++、Java 不同的地方。

package main

import "fmt"
import "runtime"

func main() {
	fmt.Print("Go runs on ")
	os := runtime.GOOS
	switch os {
	case "darwin":
		fmt.Println("OS X.")
	case "linux":
		fmt.Println("Linux.")
	default:
		// freebsd, openbsd,
		// plan9, windows...
		fmt.Printf("%s.\n", os)
	}
}

没有条件的 swaitch 等价于 switch true

package main

import (
	"fmt"

		fmt.Println("Good morning!")

	}

函数的入参与返回值

传入两个参数，一个返回值的函数写法

package main

import "fmt"

func main() {
	a, b := 1, 2
	result := add(a, b)

	return x + y

如果有多个入参是相同的类型，可以写在一起

package main

import "fmt"

func main() {
	a, b := 1, 2
	c := "I'm an add function"
	result := add(a, b, c)
	fmt.Println(result)
}

func add(x, y int, s string) int {
	fmt.Println(s, ":")
	return x + y
}

返回值也可以有多个

package main
